drop dataverse ProductReviews if exists;
create dataverse ProductReviews;
use dataverse ProductReviews;

create type Review {
    id: uuid,
    reviewerId: string,
    asin: string,
    reviewerName: string,
    helpful: {{int}},
    reviewText: string,
    overall: float,
    summary: string,
    reviewTime: date
}

create dataset Reviews(Review)
    primary key id;

for $ds in dataset Metadata.Dataset return $ds;
for $ix in dataset Metadata.Index return $ix;
